"portrait" : "landscape", unit: "mm", format: "a4" }); const mapEl = document.getElementById("map"); const originalHeight = mapEl.style.height; const originalView = map.getCenter(); const originalZoom = map.getZoom(); const bounds = L.latLngBounds(points); const pdfWidth = pdf.internal.pageSize.getWidth() - 20; const pdfHeight = pdf.internal.pageSize.getHeight() - 50; // Karte vergrößern für bessere Qualität mapEl.style.height = (isMobile ? pdfHeight * window.devicePixelRatio : pdfHeight * 5) + "px"; map.invalidateSize(); map.fitBounds(bounds, { padding: [80, 80] }); map.invalidateSize(); await new Promise(r => setTimeout(r, 1000)); try { // Zoom-/Geocoder-Steuerelemente ausblenden const controlsToHide = document.querySelectorAll('.leaflet-control-zoom, .leaflet-control-geocoder'); controlsToHide.forEach(el => { if (el) el.style.display = 'none'; }); // Map als Canvas const canvas = await html2canvas(mapEl, { useCORS: true, scale: 1, scrollY: -window.scrollY }); controlsToHide.forEach(el => { if (el) el.style.display = ''; }); const imgData = canvas.toDataURL("image/png"); let imgWidth = pdfWidth; let imgHeight = (canvas.height / canvas.width) * imgWidth; if (imgHeight > pdfHeight - 40) { const scale = (pdfHeight - 40) / imgHeight; imgWidth *= scale; imgHeight *= scale; } const yStart = 25; // Startpunkt unter Logos // --- Logos proportional laden --- async function loadLogo(src, maxWidth, maxHeight) { return new Promise(resolve => { const img = new Image(); img.onload = () => { let w = img.width, h = img.height; const scale = Math.min(maxWidth/w, maxHeight/h); w *= scale; h *= scale; const c = document.createElement("canvas"); c.width = w; c.height = h; c.getContext("2d").drawImage(img, 0, 0, w, h); resolve({ dataUrl: c.toDataURL("image/png"), width: w, height: h }); }; img.src = src; }); } const maxLogoWidth = 30, maxLogoHeight = 15; const landeslogo = await loadLogo("landeslogo.png", maxLogoWidth, maxLogoHeight); const zivilschutzlogo = await loadLogo("zivilschutzlogo.png", maxLogoWidth, maxLogoHeight); const margin = 10; pdf.addImage(landeslogo.dataUrl, "PNG", margin, 10, landeslogo.width, landeslogo.height); pdf.addImage(zivilschutzlogo.dataUrl, "PNG", pdf.internal.pageSize.getWidth()-margin-zivilschutzlogo.width, 10, zivilschutzlogo.width, zivilschutzlogo.height); // Überschrift mittig zwischen Logos const header = `Tirol - Viehbergung: ${type}`; const textWidth = pdf.getTextWidth(header); const centerX = (pdf.internal.pageSize.getWidth() - textWidth)/2; const headerY = 10 + Math.max(landeslogo.height, zivilschutzlogo.height)/2 + 5; pdf.setFontSize(12); pdf.text(header, centerX, headerY); // Map ins PDF einfügen pdf.addImage(imgData, "PNG", 10, yStart, imgWidth, imgHeight); // Text unter der Karte const yText = yStart + imgHeight + 10; pdf.setFontSize(11); pdf.text(`Aufnahmeort: ${points[0].lat.toFixed(5)}, ${points[0].lng.toFixed(5)} WGS84-Decimal`, 10, yText); pdf.text(`Ablageort: ${points[1].lat.toFixed(5)}, ${points[1].lng.toFixed(5)} WGS84-Decimal`, 10, yText+8); const now = new Date(); pdf.text(`Datum: ${now.toLocaleDateString()} ${now.toLocaleTimeString()}`, 10, yText+16); pdf.save(`viehbergung_tirol_${type}.pdf`); } catch (err) { console.error(err); alert("Fehler beim Erstellen des PDFs. Details siehe Konsole."); } finally { mapEl.style.height = originalHeight; map.invalidateSize(); map.setView(originalView, originalZoom); } };
